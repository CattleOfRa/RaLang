class HelloWorld			// Optional
					// If absent the name of the file will
					// be used instead

define main():				// Defining main is optional
	print "Hello World"		// If absent the main function will be
					// added automatically

-----

print "Hello World"			// Acceptable

-----

// Nothing will happen, since the main method is not created

define sumAndPrint(int a, int b):	// returns void
	print a + b			// Shows (a + b)

-----

// Nothing will happen, since the main method is not created

define sumAndReturn(int a, int b, int):	// returns an int
	return a + b

-----

// To call a method, the main method must be created

define main():
	print sum(2, 3)			// Shows 5

define sum(int a, int b, int):
	return a + b			// Returns 5

-----

// Won't work, as main method cannot return values

define main(int a, int b, int):
	return a + b

-----
